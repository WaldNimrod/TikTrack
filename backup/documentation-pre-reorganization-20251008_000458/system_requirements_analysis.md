## 🔍 **אלגוריתם זיהוי דרישות**

### **שלב 1: זיהוי סוג העמוד**
```javascript
function identifyPageType(pageName, pageContent) {
    // בדיקת מאפיינים
    if (hasTables(pageContent)) return 'CRUD';
    if (hasCharts(pageContent)) return 'CHARTS';
    if (hasSettings(pageContent)) return 'SETTINGS';
    if (hasAlerts(pageContent)) return 'ALERTS';
    if (hasExternalData(pageContent)) return 'EXTERNAL_DATA';
    if (hasFileManagement(pageContent)) return 'FILE_MANAGEMENT';
    if (hasDateTools(pageContent)) return 'DATE_TOOLS';
    return 'BASIC';
}
```

### **שלב 2: הקצאת מערכות לפי סוג**
```javascript
function getRequiredSystems(pageType) {
    const basePackage = ['unified-init', 'notifications', 'modals', 'section-state', 'translation', 'page-state', 'confirm-replacement', 'favicon', 'cache-sync', 'unified-cache'];
    
    switch(pageType) {
        case 'CRUD':
            return [...basePackage, 'tables', 'table-mapping', 'header', 'filters', 'category-detection'];
        case 'CHARTS':
            return [...basePackage, 'charts', 'pages', 'tables'];
        case 'SETTINGS':
            return [...basePackage, 'preferences', 'header'];
        case 'ALERTS':
            return [...basePackage, 'advanced-notifications', 'tables', 'table-mapping', 'header', 'filters'];
        case 'EXTERNAL_DATA':
            return [...basePackage, 'external-data', 'tables', 'storage'];
        case 'FILE_MANAGEMENT':
            return [...basePackage, 'file-mapping', 'header'];
        case 'DATE_TOOLS':
            return [...basePackage, 'date-utils', 'header'];
        default:
            return basePackage;
    }
}
```

### **שלב 3: בדיקת יישום**
```javascript
function checkSystemImplementation(pageName, requiredSystems) {
    const results = {};
    
    requiredSystems.forEach(system => {
        results[system] = {
            required: true,
            implemented: checkIfSystemLoaded(pageName, system),
            status: checkIfSystemLoaded(pageName, system) ? '✅' : '❌'
        };
    });
    
    return results;
}
```
